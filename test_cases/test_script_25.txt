// simple deadlock detection testcase
// T1 -> T2 |  T2 -> T3 |  T3 -> T4 | T4 -> T1
// T4 will be abort, youngest
// T1, T2, T3 will commit

begin(T1)
begin(T2)
begin(T3)
begin(T4)
R(T1,x1)
R(T2,x2)
R(T3,x3)
R(T4,x4)
W(T1,x2,200)
W(T2,x3,300)
W(T3,x4,400)
W(T4,x1,100)
end(T3)
end(T2)
end(T1)
dump()
